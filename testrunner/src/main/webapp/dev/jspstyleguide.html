Teammates JSP Style Guide

General rules
  • Separate HTML from Java
  • Try to put only presentation logic in JSP
  • Favor HTML in Java handler classes over Java in JSPs
  • Don't overuse Java code in HTML pages
  • Use JSP comments in most cases
  • Indentations should be filled with space characters
Code examples
  • JSP Page Directive(s)
    <%@ page session="false" %>
<%@ page import="java.util.*" %>
<%@ page errorPage="/common/errorPage.jsp" %>
    Or 
    <%@ page    session="false" 
            import="java.util.*"
            errorPage="/common/errorPage.jsp" 
%>
  • JSP Declaration(s)--Try not to use business logic>
    <%! 
    private int hitCount;
    private Date today; 

    public int getHitCount() {
        return hitCount;
    }
%>
    Not preferred 
    <%! private int hitCount; %>
<%! private Date today; %>
...
<%! public int getHitCount() {
        return hitCount;
    }
%>
  • Compound Indentation with JSP, HTML and Java
    <table>
    <%  if ( tableHeaderRequired ) { %>
        <tr>
            <th>Last Name</th>
            <th>First Name</th>
        </tr>
    <%  } %>
    <c:forEach var="customer" items="${customers}">
        <tr>
            <td><c:out value="${customer.lastName}"/></td>
            <td><c:out value="${customer.firstName}"/></td>
        </tr>
    </c:forEach>
</table>
    Not this 
    <table>
    <%  if ( tableHeaderRequired ) { %>
    <tr>
        <th>Last Name</th>
        <th>First Name</th>
    </tr>
    <% } %>
    <c:forEach var="customer" items="${customers}">
    <tr>
        <td><c:out value="${customer.lastName}"/></td>
        <td><c:out value="${customer.firstName}"/></td>
    </tr>
    </c:forEach>
</table>
